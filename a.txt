        function generateTOC(contentEl) {

                heading.setAttribute('id', id);

                const li = document.createElement('li');
                const a = document.createElement('a');

                a.textContent = heading.textContent;
                a.className = `toc-${heading.tagName.toLowerCase()}`;
                a.href = `#${id}`;
